CREATE DATABASE movies;
USE movies;
CREATE TABLE movies (
    id INT AUTO_INCREMENT, 
    title VARCHAR(255),
    cover VARCHAR(255), 
    description VARCHAR(1024), 
    release_date INT, 
    actors VARCHAR(255), 
    type VARCHAR(10),
    PRIMARY KEY (id) 
);
CREATE TABLE ratings (
    id INT AUTO_INCREMENT, 
    user_id INT,
    movie_id INT, 
    rating INT, 
    PRIMARY KEY (id),
    FOREIGN KEY (movie_id) REFERENCES movies(id)
);
